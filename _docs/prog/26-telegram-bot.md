    url = 'https://api.ratesapi.io/api/latest'
    params = {'base': currency_name, 'symbols': 'RUB'}
    req = requests.get(url, params=params)

    message = '–ö—É—Ä—Å {} <-> RUB —Å–µ–π—á–∞—Å *{}*'.format(currency, rate)
